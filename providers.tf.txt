# main.tf - Archivo de configuración principal de Terraform

# Proveedor Docker
provider "docker" {
  host = "tcp://localhost:2375/"
}

# Recurso Docker: Contenedor Jenkins
resource "docker_container" "jenkins_container" {
  name  = var.jenkins_container_name
  image = var.jenkins_image
  ports {
    internal = 8080
    external = 8080
  }
}

# Recurso Docker: Contenedor Docker in Docker
resource "docker_container" "dind_container" {
  name  = var.dind_container_name
  image = var.dind_image
  ports {
    internal = 2375
    external = 2375
  }
}
